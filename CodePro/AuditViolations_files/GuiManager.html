365       if (!interactionAllowed) return;

368       deselectTower(activeIndicator, activeTower,

371             centered.getYCenter(), radius);

373       myUpgradePanel.setCurrentTower(towerMap.get(towerName), tower,
374             activeIndicator);
375       myGameWorld.getMap().getChildren().add(activeIndicator);
376       tower.setOnMouseClicked(event -> deselectTower(activeIndicator, tower,
377             towerName));
